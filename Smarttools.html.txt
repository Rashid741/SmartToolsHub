// Sidebar navigation
function showTool(toolId){document.querySelectorAll(".tool").forEach(t=>t.classList.remove("active"));document.getElementById(toolId).classList.add("active");}

// Animate calculator icon
function animateCalc(id){let icon=document.getElementById(id);icon.style.display="inline-block";icon.classList.add("rotate");setTimeout(()=>{icon.style.display="none";icon.classList.remove("rotate");},1500);}

// Age Calculator
function calculateAge(){const dob=document.getElementById("dob").value;if(!dob)return;const today=new Date();const birth=new Date(dob);let years=today.getFullYear()-birth.getFullYear();let months=today.getMonth()-birth.getMonth();let days=today.getDate()-birth.getDate();if(days<0){months--;days+=30;}if(months<0){years--;months+=12;}document.getElementById("ageResult").innerText=`You are ${years} years, ${months} months, ${days} days old.`;animateCalc("calcIconAge");}

// BMI
function calculateBMI(){let height=0;if(document.getElementById("cm").value){height=parseFloat(document.getElementById("cm").value)/100;}else{let feet=parseInt(document.getElementById("feet").value)||0;let inches=parseInt(document.getElementById("inches").value)||0;height=((feet*12)+inches)*0.0254;}let weight=parseFloat(document.getElementById("kg").value)||(parseFloat(document.getElementById("lbs").value)*0.4536)||0;if(!height||!weight)return;let bmi=(weight/(height*height)).toFixed(2);document.getElementById("bmiResult").innerText=`Your BMI is ${bmi}`;animateCalc("calcIconBMI");}

// GPA
const gradePoints={"A":4.0,"A-":3.7,"B+":3.3,"B":3.0,"B-":2.7,"C+":2.3,"C":2.0,"C-":1.7,"D":1.0,"F":0.0};
function addSubject(){const div=document.createElement("div");div.innerHTML=`<select class="grade"><option value="">Grade</option><option>A</option><option>A-</option><option>B+</option><option>B</option><option>B-</option><option>C+</option><option>C</option><option>C-</option><option>D</option><option>F</option></select> <input type="number" placeholder="Credits" class="credits"> <button onclick="this.parentElement.remove()">âŒ</button>`;document.getElementById("subjects").appendChild(div);}
function calculateGPA(){const grades=[...document.querySelectorAll(".grade")];const credits=[...document.querySelectorAll(".credits")].map(c=>parseInt(c.value)||0);let totalPoints=0,totalCredits=0;grades.forEach((g,i)=>{if(g.value&&credits[i]>0){totalPoints+=gradePoints[g.value]*credits[i];totalCredits+=credits[i];}});if(totalCredits===0)return;const semGPA=(totalPoints/totalCredits).toFixed(2);const prevCgpa=parseFloat(document.getElementById("prevCgpa").value)||0;const prevCredits=parseInt(document.getElementById("prevCredits").value)||0;let newCgpa=semGPA;if(prevCgpa>0&&prevCredits>0){const totalQuality=(prevCgpa*prevCredits)+(semGPA*totalCredits);const totalAll=prevCredits+totalCredits;newCgpa=(totalQuality/totalAll).toFixed(2);}let emoji=semGPA<2.5?" ğŸ˜¢":" â¤ï¸";document.getElementById("gpaResult").innerHTML=`Semester GPA: ${semGPA}${emoji}<br>Updated CGPA: ${newCgpa}`;animateCalc("calcIconGPA");}

// Savings
function calculateSavings(){let p=parseFloat(document.getElementById("principal").value)||0;let m=parseFloat(document.getElementById("monthly").value)||0;let r=(parseFloat(document.getElementById("rate").value)||0)/100/12;let t=(parseFloat(document.getElementById("years").value)||0)*12;let future=(p*Math.pow(1+r,t))+m*((Math.pow(1+r,t)-1)/r);document.getElementById("savingsResult").innerText="Future Savings: "+future.toFixed(2);animateCalc("calcIconSavings");}

// Insurance
function calculateInsurance(){let age=parseInt(document.getElementById("ageIns").value)||30;let smoker=document.getElementById("smoker").value;let coverage=parseInt(document.getElementById("coverage").value)||100000;let factor=smoker==="Smoker"?1.5:1;let premium=(coverage/1000)*age*0.5*factor;document.getElementById("insuranceResult").innerText="Estimated Premium: $"+premium.toFixed(2)+" per year";animateCalc("calcIconInsurance");}
